		effect = {
			if = {
				limit = { has_global_flag = Ccw }

						limit = {
							NOT = { has_dynamic_modifier = { modifier = CW_ruins } }
							NOT = { is_owned_by = ROOT }
							is_core_of = CHI
						}
				add_dynamic_modifier = { modifier = CW_ruins }	
					}
				every_owned_state = {
					limit = {
						NOT = { has_dynamic_modifier = { modifier = CW_ruins } }
						NOT = { is_controlled_by = ROOT }
						is_core_of = CHI
					}

					}

			}
		}

			effect = {
				if = {
					limit = { 
						has_global_flag = Ccw 
						FROM = { tag = DSC }
						NOT = { country_exists = CHI }
						ROOT = {
								original_tag = CHI
								has_cosmetic_tag = CHI_REB_S
						}
					}
					FROM = {
						annex_country = {
							target = ROOT
							transfer_troops = yes
						}
					}
				}

				else_if = {
					limit = { 
						has_global_flag = Ccw 
						FROM = { OR = { tag = DSC tag = CHI } has_war_with = ROOT }
						DSC = { has_war_with = ROOT } CHI = { has_war_with = ROOT }
						ROOT = {
								original_tag = CHI
								has_cosmetic_tag = CHI_REB_S
						}
					}
					DSC = {
						every_controlled_state = {
							transfer_state_to = DSC
						}
					}
					CHI = {
						annex_country = {
							target = ROOT
							transfer_troops = no
						}
					}
				}
			}
		}

	on_capitulation_immediate = {
		effect = {
			if = {
					limit = {
						FROM = { tag = XIN }
						NOT = { country_exists = CHI }

						has_global_flag = cw10_already

					}

						every_owned_state = { extrameal_islam_leavel_down = yes extrameal_islam_leavel_down = yes }

							target = ROOT
							transfer_troops = no
						}
					}
			}
			else_if = {

					limit = {
						FROM = { tag = ETK }
						NOT = { country_exists = CHI }
						ROOT = { tag = XIN }
					}
					FROM = {
						annex_country = {
							target = ROOT
							transfer_troops = no
						}
					}
			
			}
