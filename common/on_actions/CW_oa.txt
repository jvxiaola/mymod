
on_actions = {


	on_weekly_CHI = {
		effect = {
			if = {
				limit = {
					ROOT = {
						has_country_flag = Ccw
					}
					DSC = {
						has_country_flag = Ccw
					}
					HKG = {
						has_country_flag = Ccw
					}
					every_country = {
						limit = {
							original_tag = CHI
							has_cosmetic_tag=CHI_REB_S
						}
						has_country_flag = Ccw
					}
				}
				every_controlled_state = { 
						limit = {
							is_CHI_CONTROL = yes
							NOT = { has_dynamic_modifier = { modifier = CW_ruins } }
							NOT = { is_owned_by = ROOT }
					} 
					add_dynamic_modifier = { modifier = CW_ruins }	
				}
				every_controlled_state = { 
					limit = {
						is_CHI_CONTROL = yes
						has_dynamic_modifier = { modifier = CW_ruins } 
							} 
						add_to_array = {
							array = global.ruined_state
						}
					}
				CHI_ruin_industry_add = yes
			}
		}
	}
